# Inferring a Rate
model{
  #per person priors
  for (cop in 1:PD){
    for (com in 1:CITY){
      
      #rate of violent arrests
      theta[com,cop] ~ dunif(0,1)

      #prior on number of violent arrests
      y[cop,com] ~ dbin(theta[com,cop], N[cop])
    }

    #Prior on the number of community interactions
    #lambda[cop] ~ dunif(30,100)
    #N[cop] ~ dpois(lambda[cop])
    N[cop] ~ dcat(p[cop,])

    for (i in 1:nmax){
      p[cop, i] <- 1/nmax
    } 
    
  }
  
}
