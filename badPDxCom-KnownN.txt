# Inferring a Rate
model{
  #per person priors
  for (cop in 1:PD){
    for (com in 1:CITY){
      sigma[cop,com] ~ dunif(0,1)
      #alpha[cop,com] ~ dunif(0,1)
      theta[com,cop] ~ dnorm(delta[com]+alpha[cop], pow(sigma[cop,com],-2))T(0,1)
      y[cop,com] ~ dbin(theta[com,cop], N[cop, com])
    }    
  }

  for (cop in 1:PD){
    alpha[cop] ~ dunif(0,1)
  } 
  
  for (com in 1:CITY){ 
    delta[com] ~ dunif(-1,1)
  }
}
