# Inferring a Rate
model{
  #per person priors
  for (cop in 1:PD){
    for (com in 1:CITY){
      theta[com,cop] ~ dnorm(delta, sigma[cop])T(0,1)
      y[cop,com] ~ dbin(theta[com,cop], N[cop])
    }

    #Prior on the number of 
    # community interactions
    #lambda[cop] ~ dunif(30,100)
    #N[cop] ~ dpois(lambda[cop])
    N[cop] ~ dcat(p[cop,])

    for (i in 1:nmax){
      p[cop, i] <- 1/nmax
    } 
    
    sigma[cop] ~ dunif(0,1)
  }
  delta ~ dunif(0,1)
}
